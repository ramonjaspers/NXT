33 void Grid::DijkstraComputePaths(vertex_t source) {

36  this->min_distance.resize(n, max_weight);

38  this->min_distance[source] = 0;
39  this->previous.clear();
40  this->previous.resize(n, -1);
41  std::set <std::pair<weight_t, vertex_t>> vertex_queue;
42  vertex_queue.insert(std::make_pair(this->min_distance[source], source));
43 
44  while (!vertex_queue.empty()) {
45  std::cout << "vertex queue size: " << vertex_queue.size() << "\n";
46  weight_t dist = vertex_queue.begin()->first;
47  vertex_t u = vertex_queue.begin()->second;
48  vertex_queue.erase(vertex_queue.begin());
49 
50  // Visit each edge exiting u
51  const std::vector <neighbor> &neighbors = this->adjacency_list[u];
52  for (std::vector<neighbor>::const_iterator neighbor_iter = neighbors.begin();
53  neighbor_iter != neighbors.end();
54  neighbor_iter++) {
55  vertex_t v = neighbor_iter->target;
56  weight_t weight = neighbor_iter->weight;
57  std::cout << neighbor_iter->target << " - " << neighbor_iter->weight << std::endl;
58  weight_t distance_through_u = dist + weight;
59  if (distance_through_u < this->min_distance[v]) {
60  vertex_queue.erase(std::make_pair(this->min_distance[v], v));
61 
62  this->min_distance[v] = distance_through_u;
63  this->previous[v] = u;
64  vertex_queue.insert(std::make_pair(this->min_distance[v], v));
65  }
66  }
67  }
68 }
